En Python

Sys.setenv(RETICULATE_PYTHON = "/Python38")
from scipy.stats import uniform
import matplotlib.pyplot as plt
import numpy as np

a = 0
b = 4
loc = a
scale = b - a

fig, ax = plt.subplots(1,1)

rv = uniform(loc = loc, scale = scale)
x = np.linspace(-0.1, 4.1)
r = rv.rvs(size = 100000)

mean, var, skew, kurt = rv.stats(moments = 'mvsk')

print(f"Media {mean}")
## Media 2.0
print(f"Varianza {var}")
## Varianza 1.3333333333333333
print(f"Sesgo {skew}")
## Sesgo 0.0
print(f"Curtosis {kurt}")
## Curtosis -1.2
ax.plot(x, rv.pdf(x), 'k-', lw = 2, label = "Funci√≥n de Densidad de U(0,4)")
ax.hist(r, density = True, histtype = "stepfilled", alpha = 0.25)
